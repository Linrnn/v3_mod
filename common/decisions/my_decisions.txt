cheat_set_modifier_1 =
{
	#is_shown =
	#{
	#}
	possible =
	{
		OR =
		{
			is_player = yes
			NOT = { has_modifier = my_modifier_1 }
		}
	}
	when_taken =
	{
		IF =
		{
			LIMIT = { has_modifier = my_modifier_1 }
			remove_modifier = my_modifier_1
		}
		ELSE =
		{
			add_modifier = my_modifier_1
		}
	}
	ai_chance =
	{
		base = 0
		modifier =
		{
			trigger =
			{
				NOT = { has_modifier = my_modifier_limit_ci }
			}
			add = 100
		}
	}
}

cheat_annex_not_decentralized_country
{
	possible =
	{
		is_player = yes
	}
	when_taken =
	{
		every_country =
		{
			IF =
			{
				LIMIT =
				{
					NOT = { THIS = ROOT }
					NOT = { is_country_type = decentralized } # 不是松散联邦（即不是可殖民国家）
				}
				ROOT = { annex = PREV }
			}
		}
	}
	ai_chance =
	{
		base = 0
	}
}

cheat_remove_build
{
	possible =
	{
		is_player = yes

			remove_building = building_shipyards # 造船厂
