# 参考mod：可控制的私人建设 Controllable Private Construction

# 煤矿
INJECT_OR_CREATE:building_coal_mine = {
	can_build_government = {
		zw_trigger_gov_investment = {
			building_1 = building_coal_mine
			bgs_1 = industry
		}
	}
	can_build_private = {
		zw_trigger_autonom_investment_manufacture = {
			building_1 = building_coal_mine
		}
	}
}
